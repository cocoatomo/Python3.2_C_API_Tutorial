    PyObject_HEAD
    PyObject *im_func;   /* The callable object implementing the method */
    PyObject *im_self;   /* The instance it is bound to */

} PyMethodObject;

#define PyMethod_Check(op) ((op)->ob_type == &PyMethod_Type)
